    \hypertarget{fundamental-equations-for-poisson-mgwr}{%
\subsubsection{Fundamental equations for Poisson
MGWR}\label{fundamental-equations-for-poisson-mgwr}}

    \begin{align}
y = Poisson[E_i exp ({\sum} {\beta} & _k x _{k,i})]   - (1) \\
E_i = Offset   - (2) \\
\hat{y} = predicted(y)   - (3) \\
z = ({\sum} {\beta} & _k x _{k,i}) + ((y - \hat{y})/\hat{y})   - (4) \\
\end{align}
